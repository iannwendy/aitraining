% Figure 1.1: Conceptual Framework
% Three-layer diagram linking clinical screening to social media signal to ML pipeline

\documentclass[border=5pt]{standalone}
\usepackage[utf8]{inputenc}
\usepackage{tikz}
\usetikzlibrary{shapes.geometric, arrows, positioning, calc, shadows.blur}

\begin{document}
\begin{tikzpicture}[
    node distance=1.2cm and 0.8cm,
    box/.style={
        rectangle,
        rounded corners=3pt,
        draw=#1,
        fill=#1!15,
        line width=1pt,
        minimum width=4cm,
        minimum height=1cm,
        align=center,
        font=\footnotesize
    },
    arrow/.style={
        ->,
        >=stealth,
        line width=1.5pt
    },
    layer/.style={

        rounded corners=5pt,

        fill=#1!10,
        line width=1.5pt,
        minimum width=14cm,
        minimum height=2.5cm,
        align=center,
        font=\large\bfseries
    }
]

% Layer 1: Clinical Reality (vermillion)
\node[layer=vermilion, label={[vermilion!80!black]north:Layer 1: Clinical Reality (DSM-5, PHQ-9)}]
    (layer1) at (0,0) {
        \begin{tabular}{ccc}
            \node[box=vermilion] {Major Depressive\\Disorder} &

    };

        \begin{tabular}{c}
            \node[box=blue] {125,329 Vietnamese YouTube Comments} \\

\node[layer=green!70!black, yshift=-8cm, label={[green!70!black]north:Layer 3: Machine Learning Pipeline (This Work)}]

        \begin{tabular}{ccccc}

            \node[box=green] {Blind\\Annotation} &

% Arrows between layers
\draw[arrow, vermillion] (layer1) -- (layer2)
